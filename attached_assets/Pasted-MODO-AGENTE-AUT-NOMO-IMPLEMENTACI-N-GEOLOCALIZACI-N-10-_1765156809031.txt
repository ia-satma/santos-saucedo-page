[TAREA 2: CREAR SERVICIO DE GEOLOCALIZACIÓN]

Crear src/services/geolocation.js:

```javascript
const SUPPORTED_LANGUAGES = ['es', 'en', 'de', 'fr', 'it', 'zh', 'ja', 'ko', 'ru', 'ar'];
const DEFAULT_LANGUAGE = 'en';

const countryToLanguage = {
  // Español
  MX: 'es', ES: 'es', AR: 'es', CO: 'es', PE: 'es', CL: 'es', 
  EC: 'es', VE: 'es', GT: 'es', CU: 'es', BO: 'es', DO: 'es',
  HN: 'es', PY: 'es', SV: 'es', NI: 'es', CR: 'es', PA: 'es', UY: 'es',
  
  // English (incluye países sin idioma soportado)
  US: 'en', GB: 'en', AU: 'en', CA: 'en', NZ: 'en', IE: 'en',
  ZA: 'en', SG: 'en', PH: 'en', IN: 'en', PK: 'en', NG: 'en',
  BR: 'en', PT: 'en', // Brasil y Portugal → English (NO hay portugués)
  
  // Deutsch
  DE: 'de', AT: 'de', CH: 'de', LI: 'de', LU: 'de',
  
  // Français
  FR: 'fr', BE: 'fr', MC: 'fr', SN: 'fr', CI: 'fr', ML: 'fr',
  
  // Italiano
  IT: 'it', SM: 'it', VA: 'it',
  
  // 中文 Chinese
  CN: 'zh', TW: 'zh', HK: 'zh', MO: 'zh',
  
  // 日本語 Japanese
  JP: 'ja',
  
  // 한국어 Korean
  KR: 'ko',
  
  // Русский Russian
  RU: 'ru', BY: 'ru', KZ: 'ru', UA: 'ru',
  
  // العربية Arabic
  AE: 'ar', SA: 'ar', EG: 'ar', IQ: 'ar', JO: 'ar', KW: 'ar',
  LB: 'ar', LY: 'ar', MA: 'ar', OM: 'ar', QA: 'ar', SY: 'ar',
  TN: 'ar', YE: 'ar', BH: 'ar', DZ: 'ar', SD: 'ar', PS: 'ar'
};

export const detectUserLanguage = async () => {
  const savedLanguage = localStorage.getItem('userLanguage');
  if (savedLanguage && SUPPORTED_LANGUAGES.includes(savedLanguage)) {
    return savedLanguage;
  }

  try {
    const response = await fetch('https://ipapi.co/json/');
    const data = await response.json();
    const countryCode = data.country_code;
    
    if (countryCode && countryToLanguage[countryCode]) {
      return countryToLanguage[countryCode];
    }
  } catch (error) {
    console.warn('Geolocation failed, using browser language');
  }

  const browserLang = navigator.language?.split('-')[0];
  if (browserLang && SUPPORTED_LANGUAGES.includes(browserLang)) {
    return browserLang;
  }

  return DEFAULT_LANGUAGE;
};

export const isRTL = (languageCode) => {
  return languageCode === 'ar';
};
```

[TAREA 4: CREAR CONFIGURACIÓN i18n]

Crear src/i18n.js:

```javascript
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import { detectUserLanguage, isRTL } from './services/geolocation';

import esTranslations from './locales/es/common.json';
import enTranslations from './locales/en/common.json';
import deTranslations from './locales/de/common.json';
import frTranslations from './locales/fr/common.json';
import itTranslations from './locales/it/common.json';
import zhTranslations from './locales/zh/common.json';
import jaTranslations from './locales/ja/common.json';
import koTranslations from './locales/ko/common.json';
import ruTranslations from './locales/ru/common.json';
import arTranslations from './locales/ar/common.json';

const resources = {
  es: { translation: esTranslations },
  en: { translation: enTranslations },
  de: { translation: deTranslations },
  fr: { translation: frTranslations },
  it: { translation: itTranslations },
  zh: { translation: zhTranslations },
  ja: { translation: jaTranslations },
  ko: { translation: koTranslations },
  ru: { translation: ruTranslations },
  ar: { translation: arTranslations }
};

export const initI18n = async () => {
  const detectedLanguage = await detectUserLanguage();
  
  await i18n
    .use(initReactI18next)
    .init({
      resources,
      lng: detectedLanguage,
      fallbackLng: 'en',
      supportedLngs: ['es', 'en', 'de', 'fr', 'it', 'zh', 'ja', 'ko', 'ru', 'ar'],
      interpolation: { escapeValue: false }
    });

  document.documentElement.lang = detectedLanguage;
  document.documentElement.dir = isRTL(detectedLanguage) ? 'rtl' : 'ltr';
  
  return i18n;
};

export { isRTL };
export default i18n;
```

[TAREA 7: CREAR LANGUAGE SWITCHER]

Crear src/components/LanguageSwitcher.jsx:

```jsx
import React from 'react';
import { useTranslation } from 'react-i18next';
import { isRTL } from '../services/geolocation';

const languages = [
  { code: 'es', name: 'Español', flag: '🇲🇽' },
  { code: 'en', name: 'English', flag: '🇺🇸' },
  { code: 'de', name: 'Deutsch', flag: '🇩🇪' },
  { code: 'fr', name: 'Français', flag: '🇫🇷' },
  { code: 'it', name: 'Italiano', flag: '🇮🇹' },
  { code: 'zh', name: '中文', flag: '🇨🇳' },
  { code: 'ja', name: '日本語', flag: '🇯🇵' },
  { code: 'ko', name: '한국어', flag: '🇰🇷' },
  { code: 'ru', name: 'Русский', flag: '🇷🇺' },
  { code: 'ar', name: 'العربية', flag: '🇦🇪' }
];

const LanguageSwitcher = () => {
  const { i18n } = useTranslation();
  const changeLanguage = (langCode) => {
    i18n.changeLanguage(langCode);
    localStorage.setItem('userLanguage', langCode);
    document.documentElement.lang = langCode;
    document.documentElement.dir = isRTL(langCode) ? 'rtl' : 'ltr';
  };
  return (
    <select value={i18n.language} onChange={(e) => changeLanguage(e.target.value)} aria-label="Select language">
      {languages.map((lang) => (
        <option key={lang.code} value={lang.code}>{lang.flag} {lang.name}</option>
      ))}
    </select>
  );
};

export default LanguageSwitcher;
```
